import random

# Step 1: Define fuzzy membership functions
def fuzzy_distance(d):
    if d < 3:
        return {"near": 1, "medium": 0, "far": 0}
    elif 3 <= d <= 6:
        return {"near": (6 - d) / 3, "medium": (d - 3) / 3, "far": 0}
    else:
        return {"near": 0, "medium": 0, "far": 1}

def fuzzy_speed(s):
    if s < 3:
        return {"slow": 1, "medium": 0, "fast": 0}
    elif 3 <= s <= 6:
        return {"slow": (6 - s) / 3, "medium": (s - 3) / 3, "fast": 0}
    else:
        return {"slow": 0, "medium": 0, "fast": 1}

# Step 2: Define fuzzy rules
def fuzzy_decision(distance, speed):
    near, medium, far = distance["near"], distance["medium"], distance["far"]
    slow, med, fast = speed["slow"], speed["medium"], speed["fast"]

    jump_low = medium * slow
    jump_med = (medium * med) + (far * slow)
    jump_high = (near * fast) + (near * med)

    max_val = max(jump_low, jump_med, jump_high)
    if max_val == jump_high:
        return "High Jump ðŸ¦˜"
    elif max_val == jump_med:
        return "Medium Jump ðŸ‡"
    else:
        return "Low Jump ðŸ¢"

# Step 3: Run simulation for 25 targets
for i in range(1, 26):
    distance = random.uniform(1, 10)
    speed = random.uniform(1, 10)

    fuzzy_d = fuzzy_distance(distance)
    fuzzy_s = fuzzy_speed(speed)
    decision = fuzzy_decision(fuzzy_d, fuzzy_s)

    print(f"ðŸŽ Target {i}: Distance={distance:.1f}, Speed={speed:.1f} â†’ {decision}")
